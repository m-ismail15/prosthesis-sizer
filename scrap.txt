import sys
from PyQt6.QtWidgets import (
    QApplication, QWidget, QVBoxLayout, QLabel, QLineEdit, QPushButton,
    QTableWidget, QTableWidgetItem, QStackedWidget, QHBoxLayout, QTextEdit
)
from firebase_config import db
from sizing import compute_prosthesis_size

# ---------------- SAVE RECORD ----------------
def save_record(name, wrist, forearm, size, category):
    record = {
        "name": name,
        "wrist_circ": wrist,
        "forearm_circ": forearm,
        "recommended_size": size,
        "category": category
    }
    db.collection("prosthesis_records").add(record)

# ---------------- MAIN APP ----------------
class ProsthesisApp(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Prosthesis Sizing Tool")
        self.resize(700, 500)

        # Page container
        self.stack = QStackedWidget()
        layout = QVBoxLayout()
        layout.addWidget(self.stack)
        self.setLayout(layout)

        # Add pages
        self.page_home = self.build_home()
        self.page_records = self.build_records()
        self.page_guide = self.build_measurement_guide()

        self.stack.addWidget(self.page_home)
        self.stack.addWidget(self.page_records)
        self.stack.addWidget(self.page_guide)

        # Navigation buttons
        nav_layout = QHBoxLayout()
        home_btn = QPushButton("🏠 Home")
        rec_btn = QPushButton("📋 Records")
        guide_btn = QPushButton("📖 Measurement Guide")

        home_btn.clicked.connect(lambda: self.stack.setCurrentWidget(self.page_home))
        rec_btn.clicked.connect(lambda: self.stack.setCurrentWidget(self.page_records))
        guide_btn.clicked.connect(lambda: self.stack.setCurrentWidget(self.page_guide))

        nav_layout.addWidget(home_btn)
        nav_layout.addWidget(rec_btn)
        nav_layout.addWidget(guide_btn)

        layout.addLayout(nav_layout)

    # ---------------- HOME PAGE ----------------
    def build_home(self):
        page = QWidget()
        layout = QVBoxLayout()

        self.name_input = QLineEdit()
        self.wrist_input = QLineEdit()
        self.forearm_input = QLineEdit()
        self.result_label = QLabel("")

        calc_btn = QPushButton("Compute Size")
        calc_btn.clicked.connect(self.calculate)

        layout.addWidget(QLabel("Patient Name"))
        layout.addWidget(self.name_input)
        layout.addWidget(QLabel("Wrist Circumference (cm)"))
        layout.addWidget(self.wrist_input)
        layout.addWidget(QLabel("Forearm Circumference (cm)"))
        layout.addWidget(self.forearm_input)
        layout.addWidget(calc_btn)
        layout.addWidget(self.result_label)

        page.setLayout(layout)
        return page

    # ---------------- RECORDS PAGE ----------------
    def build_records(self):
        page = QWidget()
        layout = QVBoxLayout()

        self.table = QTableWidget()
        self.table.setColumnCount(5)
        self.table.setHorizontalHeaderLabels(["Name", "Wrist", "Forearm", "Size", "Category"])

        load_btn = QPushButton("Load Records")
        load_btn.clicked.connect(self.load_records)

        layout.addWidget(load_btn)
        layout.addWidget(self.table)
        page.setLayout(layout)
        return page

    def load_records(self):
        records = db.collection("prosthesis_records").stream()
        self.table.setRowCount(0)

        for i, rec in enumerate(records):
            r = rec.to_dict()
            self.table.insertRow(i)
            self.table.setItem(i, 0, QTableWidgetItem(r["name"]))
            self.table.setItem(i, 1, QTableWidgetItem(str(r["wrist_circ"])))
            self.table.setItem(i, 2, QTableWidgetItem(str(r["forearm_circ"])))
            self.table.setItem(i, 3, QTableWidgetItem(str(r["recommended_size"])))
            self.table.setItem(i, 4, QTableWidgetItem(r["category"]))

    # ---------------- MEASUREMENT GUIDE PAGE ----------------
    def build_measurement_guide(self):
        page = QWidget()
        layout = QVBoxLayout()

        guide_text = QTextEdit()
        guide_text.setReadOnly(True)
        guide_text.setText(
            """
📖 **Measurement Guide**

To ensure accurate prosthesis fitting, please take the following measurements at the specified anatomical landmarks:

1. **Wrist Circumference**  
   - Measured around the styloid process of the radius and ulna.  
   - The tape should be snug but not compressing the skin.

2. **Forearm Circumference**  
   - Taken at 11% of the forearm length from the olecranon (elbow tip) towards the wrist.  
   - Ensure the tape is perpendicular to the forearm axis.

3. **Elbow Width (optional)**  
   - Distance between the medial and lateral epicondyles of the humerus.  
   - Helps in socket design and articulation allowance.

4. **Forearm Length**  
   - From olecranon (elbow tip) to the stylion (distal end of the radius).  
   - Keep the arm slightly flexed when measuring.

⚠️ Consistency is key — always measure on the same anatomical landmarks, using flexible tape, and record in centimeters.
"""
        )

        layout.addWidget(guide_text)
        page.setLayout(layout)
        return page

    # ---------------- SIZE CALCULATION ----------------
    def calculate(self):
        try:
            name = self.name_input.text()
            wrist = float(self.wrist_input.text())
            forearm = float(self.forearm_input.text())

            size, category = compute_prosthesis_size(wrist, forearm)
            self.result_label.setText(f"Size: {size} cm ({category})")

            save_record(name, wrist, forearm, size, category)
        except ValueError:
            self.result_label.setText("⚠ Please enter valid numbers")


# ---------------- RUN APP ----------------
if __name__ == "__main__":
    app = QApplication(sys.argv)
    window = ProsthesisApp()
    window.show()
    sys.exit(app.exec())
